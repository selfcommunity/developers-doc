> The Group Info Widget component renders the group main info as: group name, description, visibility and privacy, date of creation and administrator.

[//]: # (import Logged from "../../Logged";)

[//]: # (import {GroupInfoWidget} from '@selfcommunity/react-ui';)

[//]: # ()
[//]: # (<Logged><GroupInfoWidget groupId={3}/></Logged>)


#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {GroupInfoWidget} from '@selfcommunity/react-ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><GroupInfoWidget /></SCContextProvider>
   );
 }
```
