> The Thread component renders the conversation between users and it is composed of different message items that contains<br/>
 the body of the message related to their type("text", "document", "image" or "video").

import Thread from './thread.js';
import Logged from "../../Logged";

<Logged><Thread/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {Thread} from '@selfcommunity/react-ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><Thread /></SCContextProvider>
   );
 }
```
