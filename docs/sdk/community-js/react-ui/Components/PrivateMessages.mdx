> The Private Messages template renders private messages page.

import PrivateMessages from './privateMessages.js';
import Logged from "../../Logged";

[//]: # (<Logged><PrivateMessages/></Logged>)

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {PrivateMessageComponent} from '@selfcommunity/react-ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><PrivateMessageComponent/></SCContextProvider>
   );
 }
 ```
