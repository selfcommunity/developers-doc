> The Message component renders a message editor with emoji and attach media menus.

import MessageEditor from './messageEditor.js';
import Logged from "../../Logged";

<br/>
<Logged><MessageEditor/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {MessageEditor} from '@selfcommunity/react-ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><MessageEditor /></SCContextProvider>
   );
 }
```
