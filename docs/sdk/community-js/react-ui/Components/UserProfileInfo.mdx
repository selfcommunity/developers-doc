> The User Profile Info component renders some user information such its real name, job and join date.

import UserProfileInfo from './userProfileInfo.js';
import Logged from "../../Logged";

<Logged><UserProfileInfo/></Logged>


#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {UserProfileInfo} from '@selfcommunity/react-ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><UserProfileInfo /></SCContextProvider>
   );
 }
```
