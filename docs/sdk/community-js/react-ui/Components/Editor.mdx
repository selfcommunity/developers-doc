> The Editor component renders a text editor with emoji menu.

import Logged from "../../Logged";
import {Editor} from '@selfcommunity/react-ui';

<Logged><Editor/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {Editor} from '@selfcommunity/react-ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><Editor /></SCContextProvider>
   );
 }
```
