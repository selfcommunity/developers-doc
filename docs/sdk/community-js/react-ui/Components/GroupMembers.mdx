> The Group Members Widget renders the list of group members.


[//]: # (import Logged from "../../Logged";)

[//]: # (import {GroupMembersWidget} from '@selfcommunity/react-ui';)

[//]: # ()
[//]: # (<Logged><GroupMembersWidget groupId={7}/></Logged>)

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {GroupMembersWidget} from '@selfcommunity/react-ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><GroupMembersWidget /></SCContextProvider>
   );
 }
```
