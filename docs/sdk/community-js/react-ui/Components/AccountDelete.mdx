> The Account Delete component allows users to display the logic behind user deletion.<br/>
This component is composed of a list containing the instructions for deleting the account, a form in which to enter the email and a button for sending the request.

import Logged from '../../Logged.js';
import {AccountDelete} from '@selfcommunity/react-ui';

<Logged><AccountDelete/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {AccountDelete} from '@selfcommunity/react-ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><AccountDelete /></SCContextProvider>
   );
 }
```
