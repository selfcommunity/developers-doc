> The Main Feed template renders the main feed page.

import MainFeed from './mainFeed.js';
import Logged from "../../Logged";

<Logged><MainFeed/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {MainFeed} from '@selfcommunity/react-templates';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><MainFeed/></SCContextProvider>
   );
 }
 ```
