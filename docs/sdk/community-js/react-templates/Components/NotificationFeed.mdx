> The Notification Feed template renders the notification feed page.

import NotificationFeed from './notificationFeed.js';
import Logged from "../../Logged";

<Logged><NotificationFeed/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {NotificationFeed} from '@selfcommunity/react-templates';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><NotificationFeed/></SCContextProvider>
   );
 }
 ```
