> The LoyaltyProgramDetail template renders the loyalty program detail page.


import Logged from "../../Logged";
import {LoyaltyProgramDetail} from '@selfcommunity/react-templates';

<Logged><LoyaltyProgramDetail/></Logged>


#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {LoyaltyProgramDetail} from '@selfcommunity/react-templates';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><LoyaltyProgramDetail/></SCContextProvider>
   );
 }
```
