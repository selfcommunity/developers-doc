> The Feed Object Detail template renders the feed object detail page.


import Logged from "../../Logged";
import {FeedObjectDetail} from '@selfcommunity/react-templates';

<Logged><FeedObjectDetail feedObjectId={25} feedObjectType={'discussion'}/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/react-core';
 import {FeedObjectDetail} from '@selfcommunity/react-templates';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><FeedObjectDetail/></SCContextProvider>
   );
 }
 ```
