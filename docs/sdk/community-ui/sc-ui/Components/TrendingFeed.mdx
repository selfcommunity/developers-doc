> The Trending Feed component renders a list of trending posts, discussions, statuses.

import TrendingFeed from './trendingFeed.js';
import Logged from "../../Logged";
<Logged><TrendingFeed/></Logged>


#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {TrendingFeed} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><TrendingFeed /></SCContextProvider>
   );
 }
```
