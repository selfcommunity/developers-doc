> The Users Followed component renders the list of the users followed by the user.

import UsersFollowed from './usersFollowed.js';
import Logged from "../../Logged";

<Logged><UsersFollowed/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {UsersFollowed} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><UsersFollowed /></SCContextProvider>
   );
 }
```
