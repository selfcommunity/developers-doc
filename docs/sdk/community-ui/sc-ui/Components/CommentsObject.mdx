> The Comments Object component renders a list of comments.

import CommentsObject from './commentsObject.js';
import Logged from "../../Logged";

<Logged><CommentsObject/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {CommentsObject} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><CommentsObject /></SCContextProvider>
   );
 }
```
