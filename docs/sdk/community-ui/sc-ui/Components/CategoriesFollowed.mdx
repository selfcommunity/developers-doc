> The Categories Followed component is a list that renders a list of Categories followed by the authenticated user.

import Logged from "../../Logged";
import CategoriesFollowed from './categoriesFollowed.js';

<Logged><CategoriesFollowed/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {CategoriesFollowed} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><CategoriesFollowed /></SCContextProvider>
   );
 }
```
