> The Account Sign Up component allows users to create an account.<br/>
This component is composed of a simple email, username and password form and a "register" button that submits the request.

import AccountSignUp  from './accountSignUp.js';
import Anonymous from '../../Anonymous.js';

<Anonymous><AccountSignUp/></Anonymous>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {AccountSignUp} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><AccountSignUp /></SCContextProvider>
   );
 }
```
