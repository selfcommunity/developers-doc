> The Inline Composer component renders an inline composer.

import InlineComposer from './inlineComposer.js';
import Logged from "../../Logged";

<Logged><InlineComposer/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {InlineComposer} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><InlineComposer /></SCContextProvider>
   );
 }
```
