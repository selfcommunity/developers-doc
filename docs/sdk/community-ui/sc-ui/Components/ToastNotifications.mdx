> The Toast Notifications component renders  toast notifications.

import ToastNotifications from './toastNotifications.js';
import Logged from "../../Logged";

<Logged><ToastNotifications/></Logged>


#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {ToastNotifications} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><ToastNotifications /></SCContextProvider>
   );
 }
```
