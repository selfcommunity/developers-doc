> The Loyalty Program component  renders the loyalty program card.

import LoyaltyProgram from './loyaltyProgram.js';
import Logged from "../../Logged";

<Logged><LoyaltyProgram/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {LoyaltyProgram} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><LoyaltyProgram /></SCContextProvider>
   );
 }
```
