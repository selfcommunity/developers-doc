> The Platform component renders a list of related discussions.

import RelatedDiscussion from './relatedDiscussion.js';
import Logged from "../../Logged";

<Logged><RelatedDiscussion/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {RelatedDiscussion} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><RelatedDiscussion /></SCContextProvider>
   );
 }
```
