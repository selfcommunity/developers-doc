> The Composer component renders a composer.

import Composer from './composer.js';
import Logged from "../../Logged";

<Logged><Composer/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {Composer} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><Composer /></SCContextProvider>
   );
 }
```
