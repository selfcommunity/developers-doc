> The Feed Object component renders a Post, Discussion or Status.

import FeedObject from './feedObject.js';
import Logged from "../../Logged";

<Logged><FeedObject/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {FeedObject} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><FeedObject /></SCContextProvider>
   );
 }
```
