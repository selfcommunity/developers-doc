> The User component renders a user item containing user's avatar and username, info (or followers counter) and a follow button, <br/>
that allows users to follow/unfollow others.

import User from './user.js';
import Logged from "../../Logged";

<Logged><User/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {User} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><User /></SCContextProvider>
   );
 }
```
