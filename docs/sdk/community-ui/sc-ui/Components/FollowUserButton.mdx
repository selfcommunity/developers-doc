> The Follow User Button component renders a button for user following.

import FollowUserButton from './followUserButton.js';
import Anonymous from "../../Anonymous";

<br/>
<Anonymous><FollowUserButton/></Anonymous>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {FollowUserButton} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><FollowUserButton /></SCContextProvider>
   );
 }
```
