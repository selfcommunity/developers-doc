> The Follow Category Button component renders a button that allows users to follow or unfollow a category.

import FollowCategoryButton from './followCategoryButton.js';
import Anonymous from "../../Anonymous";

<br/>
<Anonymous><FollowCategoryButton/></Anonymous>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {FollowCategoryButton} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><FollowCategoryButton /></SCContextProvider>
   );
 }
```
