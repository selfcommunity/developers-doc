> The Notification component renders a list of notifications.

<!-- `import Notification from './notification.js'; -->
<!-- `import Logged from "../../Logged";` -->
<!-- `<Logged><Notification/></Logged>` -->

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {Notification} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><Notification /></SCContextProvider>
   );
 }
```
