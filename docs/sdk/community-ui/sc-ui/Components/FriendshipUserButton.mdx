> The Friendship User Button component renders the button for user friendship.

import FriendshipUserButton from './friendshipuserbutton.js';
import Anonymous from "../../Anonymous";

<br/>
<Anonymous><FriendshipUserButton/></Anonymous>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {FriendshipUserButton} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><FriendshipUserButton /></SCContextProvider>
   );
 }
```
