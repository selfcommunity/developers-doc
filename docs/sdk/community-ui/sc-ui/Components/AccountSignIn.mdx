> The Account Sign In component renders the login form.

import Anonymous from '../../Anonymous.js';
import AccountSignIn from './accountSignIn.js';

<Anonymous><AccountSignIn/></Anonymous>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {AccountSignIn} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><AccountSignIn /></SCContextProvider>
   );
 }
```
