> The Snippet Notifications component renders snippet notifications.

import SnippetNotifications from './snippetNotifications.js';
import Logged from "../../Logged";

<Logged><SnippetNotifications/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {SnippetNotifications} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><SnippetNotifications /></SCContextProvider>
   );
 }
```
