> The Categories Popular component renders a list of popular categories.

import CategoriesPopular from './categoriesPopular.js';
import Logged from "../../Logged";

<Logged><CategoriesPopular/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {CategoriesPopular} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><CategoriesPopular /></SCContextProvider>
   );
 }
```
