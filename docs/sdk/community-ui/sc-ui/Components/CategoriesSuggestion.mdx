> The Categories Suggestion component renders a list of suggested categories.

import CategoriesSuggestion from './categoriesSuggestion.js';
import Logged from "../../Logged";

<Logged><CategoriesSuggestion/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {CategoriesSuggestion} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><CategoriesSuggestion /></SCContextProvider>
   );
 }
```
