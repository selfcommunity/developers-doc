> The Trending People component renders a list of trending people.

import TrendingPeople from './trendingPeople.js';
import Logged from "../../Logged";

<br/>
<Logged><TrendingPeople/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {TrendingPeople} from '@selfcommunity/ui';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><TrendingPeople /></SCContextProvider>
   );
 }
```
