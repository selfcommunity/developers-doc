> The Explore Feed template renders the explore feed page.

import ExploreFeed from './exploreFeed.js';
import Logged from "../../Logged";

<Logged><ExploreFeed/></Logged>

#### Usage

```jsx
 import React from 'react';
 import {SCContextProvider} from '@selfcommunity/core';
 import {ExploreFeed} from '@selfcommunity/templates';

 function App() {
   const conf = {
     portal: '<community-path>',
     session: {
       type: 'OAuth',
       token: '<community-token>',
       refreshToken: '<community-refresh-token>',
       refreshTokenEndpoint: '<community-refresh-endpoint>',
     },
   };
   return (
     <SCContextProvider conf={conf}><ExploreFeed/></SCContextProvider>
   );
 }
```
